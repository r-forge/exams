%% Structure of exercise files after running Sweave on them

\begin{question}           % compulsory

  The main text of the question

  \begin{image}            % optional
    If mathematical symbols or formulas are too complex to be rendered
    as HTML this indicates to render this block as an image. Can be
    complete text of question (except answerlist). For LaTeX define as
    empty environment (i.e., ignore).
  \end{image}

  \begin{answerlist}       % only for types with more than one answer
   \item answer 1
   \item answer 2
    ...
  \end{answerlist}

\end{question}

\begin{solution}           % optional

  Optional (but of course recommended) solution

  \begin{image}            % optional
    see above
  \end{image}

  \begin{answerlist}       % only for types with more than one answer
   \item answer 1
   \item answer 2
    ...
  \end{answerlist}
  
\end{solution}
