

\begin{question}

\includegraphics[width=5cm]{automaton.png}

  \item 101000

  \item 10100
  \item 0000110
  \item 00111
\end{answerlist}
\end{question}


\begin{solution}

odd number of ones and
odd number of zeros.

\begin{answerlist}

  \item Not accepted
  \item Not accepted

  \item Not accepted
\end{answerlist}
\end{solution}
